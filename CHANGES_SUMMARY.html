  6. ServiceDetails sees autoBook and opens the BookingConfirmationFlow automatically with date/time/location pre-selected.

TypewriterText component (key parts)

const TYPING_SPEED_MS = 16;

function TypewriterText({ text, style, animate, onTick, onComplete }) {
  const [shown, setShown] = useState(animate ? '' : text);
  const lastAnimatedRef = useRef(animate ? '' : text);
  const [cursorOn, setCursorOn] = useState(true);
  const [typing, setTyping] = useState(false);

  useEffect(() => {
    if (!animate) { setShown(text); onComplete?.(); return; }
    if (lastAnimatedRef.current === text) return;
    lastAnimatedRef.current = text;
    setShown(''); setTyping(true);
    let i = 0; let timer;
    const step = () => {
      i += 1;
      setShown(text.slice(0, i));
      onTick?.();
      if (i < text.length) timer = setTimeout(step, TYPING_SPEED_MS);
      else { setTyping(false); onComplete?.(); }
    };
    timer = setTimeout(step, TYPING_SPEED_MS);
    return () => clearTimeout(timer);
  }, [text, animate]);

  return <Text style={style}>{shown}{typing && '▌'}</Text>;
}

Word-boundary regex (delimiter-safe)

function detectServiceKeyword(text = '') {
  if (!text) return null;
  const lower = text.toLowerCase();
  for (const entry of SERVICE_KEYWORDS) {
    for (const w of entry.words) {
      const re = new RegExp(`(^|\\W)${escapeForRegex(w.trim())}(\\W|$)`);
      if (re.test(lower)) return entry.label;
    }
  }
  return null;
}

const OPENROUTER_API_KEY = process.env.OPENROUTER_API_KEY || process.env.GEMINI_API_KEY;
const OPENROUTER_MODEL = process.env.OPENROUTER_MODEL || 'openai/gpt-5.3-chat';
const OPENROUTER_URL = 'https://openrouter.ai/api/v1/chat/completions';

async function callOpenRouter(prompt, { json = false } = {}) {
  if (!OPENROUTER_API_KEY) throw new Error('OPENROUTER_API_KEY is not set.');
  const body = {
    model: OPENROUTER_MODEL,
    messages: [{ role: 'user', content: prompt }],
  };
  if (json) body.response_format = { type: 'json_object' };
  const res = await fetch(OPENROUTER_URL, {
    method: 'POST',
    headers: {
      Authorization: `Bearer ${OPENROUTER_API_KEY}`,
      'Content-Type': 'application/json',
      'HTTP-Referer': 'https://madadgar.local',
      'X-Title': 'Madadgar',
    },
    body: JSON.stringify(body),
  });
  if (!res.ok) throw new Error(`OpenRouter ${res.status}: ${await res.text()}`);
  const data = await res.json();
  return data?.choices?.[0]?.message?.content || '';
}

10. Heuristic fallback when the LLM fails

Trigger
Gemini quota errors caused conversation dead-ends. Without an LLM, the chat needed to keep functioning.
File modified

Backend/agents/intentAgent.js

What was added

  • Service keyword detection — same regex map used as a guardrail above.
  • Location detection — list of Pakistani cities (Karachi, Lahore, Islamabad, …) and area hints (DHA, Clifton, G-13, …).
  • Time detection — regex patterns for today / tomorrow / weekdays / clock / morning / afternoon / evening / now / asap.
  • History-aware — only scans User: lines from the formatted history (assistant fallbacks contain "plumber" / "today" and were polluting matches).
  • Robust JSON parsing — regex-extracts the first {…} from any model output as a backstop.

Heuristic intent builder

function heuristicIntent(userInput = '', history = '') {
  const userHistory = history
    .split('\n')
    .filter((line) => /^\s*User:/i.test(line))
    .map((line) => line.replace(/^\s*User:\s*/i, ''))
    .join('\n');

  const combined = `${userHistory}\n${userInput}`;
  const service = detectServiceKeyword(userInput) || detectServiceKeyword(userHistory);
  const location =
    detectLocationKeyword(userInput) ||
    detectLocationKeyword(userHistory) ||
    detectLocationKeyword(combined);
  const time = detectTimeKeyword(userInput) || detectTimeKeyword(userHistory);

  return {
    intentType: service ? 'service_request' : 'unclear',
    service: service || null,
    location: location || null,
    time: time || null,
    language: 'english',
  };
}

  if (/\btoday\b|\bnow\b|\basap\b|\bavailable\b/.test(lower)) {
    result.dateKey = dateOptions[0]?.key;
  } else if (/\btomorrow\b/.test(lower)) {
    result.dateKey = dateOptions[1]?.key;
  } else {
    // weekday match — next occurrence
    ...
  }

  const clock = lower.match(/(\d{1,2})(?::(\d{2}))?\s*(am|pm)/);
  if (clock) {
    const hh = parseInt(clock[1], 10);
    const wanted = `${hh}:00 ${clock[3].toUpperCase()}`;
    if (TIME_SLOTS.includes(wanted)) result.timeSlot = wanted;
  } else if (/\bmorning\b/.test(lower)) result.timeSlot = '10:00 AM';
  else if (/\bnoon\b/.test(lower)) result.timeSlot = '12:00 PM';
  else if (/\bafternoon\b/.test(lower)) result.timeSlot = '2:00 PM';
  else if (/\bevening\b/.test(lower)) result.timeSlot = '6:00 PM';
  else if (/\bnight\b/.test(lower)) result.timeSlot = '8:00 PM';

  return result;
};

FadeIn wrapper + completedIds gating

function FadeIn({ children, duration = 260, style }) {
  const opacity = useRef(new Animated.Value(0)).current;
  useEffect(() => {
    Animated.timing(opacity, {
      toValue: 1, duration, useNativeDriver: true,
    }).start();
  }, []);
  return <Animated.View style={[{ opacity }, style]}>{children}</Animated.View>;
}

// In renderItem:
const extrasReady = !shouldAnimate || completedIds.has(item.id);

{extrasReady && hasSuggestions && (
  <FadeIn style={styles.suggestionsWrap}>
    {item.suggestions.map((svc) => renderSuggestionCard(svc, item))}
  </FadeIn>
)}

Every post-bubble block (intent chips, suggestion cards, slot pills, category pivots) is gated on extrasReady. TypewriterText fires onComplete → adds id to completedIds → the gate flips → blocks mount inside FadeIn and animate opacity 0→1 over 260 ms.
